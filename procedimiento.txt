DELIMITER $$
CREATE DEFINER=`root`@`localhost` PROCEDURE `sp_fotosequipo_de_usuario`()
    NO SQL
SELECT usuario.idUsuario, equipo.idEquipo, equipo_categoria.idCategoria, categoria.nombre, fotos_equipos.idFoto
FROM usuario JOIN equipo ON usuario.idEquipo=equipo.idEquipo JOIN equipo_categoria ON equipo.idEquipo=equipo_categoria.idEquipo JOIN categoria ON equipo_categoria.idCategoria=categoria.idCategoria JOIN fotos_equipos ON categoria.idCategoria=fotos_equipos.idCategoria
GROUP BY usuario.idUsuario$$
DELIMITER ;
